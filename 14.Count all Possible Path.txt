//User function template for C++
class Solution{
public:
	int rowWithMax1s(vector<vector<int> > arr, int r, int c) {
        int max_no = 0;
        int max_r = -1;
        for(int i = 0; i < r; i++)
        {
            for(int j = 0; j < c; j++)
            {
                if(arr[i][j] != 0)
                {
                    int no = c - j;
                    if(no > max_no)
                    {
                        max_no = no;
                        max_r = i;
                    }
                    break;
                }
            }
        }
        return max_r;
    }

};
